meaning.innerText=data[i].m;
roman.classList.add("hidden");
meaning.classList.add("hidden");
}

function speak(){
let msg=new SpeechSynthesisUtterance(data[i].t);
msg.lang="th-TH";
speechSynthesis.speak(msg);
}
