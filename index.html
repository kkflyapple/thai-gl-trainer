<script>
/* --- Flashcards --- */
const flashcards = [
{t:"สวัสดี", m:"Hello"},
{t:"คิดถึงนะ", m:"I miss you"},
{t:"ชอบเธอ", m:"I like you"},
{t:"ไปกินข้าวกันไหม", m:"Eat together?"},
{t:"น่ารักจัง", m:"So cute"}
];
let i = 0;
function show(){ document.getElementById("meaning").classList.remove("hidden"); }
function next(){
i = (i+1)%flashcards.length;
document.getElementById("thai").innerText = flashcards[i].t;
document.getElementById("meaning").innerText = flashcards[i].m;
document.getElementById("meaning").classList.add("hidden");
}
function speak(){
let msg = new SpeechSynthesisUtterance(flashcards[i].t);
msg.lang="th-TH";
speechSynthesis.speak(msg);
}

/* --- Dialogue scenes --- */
const scenes = [
"A: คิดถึงนะ<br>B: เราก็คิดถึง",
"A: เป็นอะไรไหม?<br>B: ไม่เป็นไร",
"A: ชอบเธอ<br>B: จริงเหรอ"
];
let s = 0;
function nextScene(){
s = (s+1)%scenes.length;
dialogue.innerHTML = scenes[s];
}

/* --- Roleplay --- */
function checkRP(){
let ans = rpInput.value.toLowerCase();
if(ans.includes("yes")||ans.includes("ok")){
  rpResult.innerText = "💕 Cute reply!";
} else {
  rpResult.innerText = "Try something softer 😊";
}
}

/* --- Quiz --- */
const quiz = [
{q:"น่ารักจัง", a:"cute"},
{q:"ชอบเธอ", a:"i like you"},
{q:"คิดถึงนะ", a:"i miss you"}
];
let qIndex = 0;
function checkQuiz(){
let user = quizA.value.toLowerCase();
if(user.includes(quiz[qIndex].a)){
  quizR.innerText = "✅ Correct!";
} else {
  quizR.innerText = `❌ ${quiz[qIndex].a}`;
}
qIndex = (qIndex+1)%quiz.length;
quizQ.innerText = quiz[qIndex].q;
quizA.value = "";
}
